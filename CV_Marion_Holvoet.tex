% Section heading for the RIGHT column — accent bar left + rule
\newcommand{\rsection}[1]{%
  \vspace{12pt}%
  \noindent{\color{accent}\vrule width 4pt height 1.15em depth 0.25ex}\hspace{8pt}{\large\bfseries\color{dark} #1}\\[-6pt]%
  {\color{accentlight}\rule{\linewidth}{0.7pt}}%
  \vspace{5pt}%
}

\newcommand{\lsection}[1]{%
  \vspace{6pt}%
  {\footnotesize\bfseries\color{accent}\MakeUppercase{#1}}\\[-3pt]%
  {\color{accentlight}\rule{\linewidth}{0.5pt}}%
  \vspace{3pt}%
}

% Experience entry
\newcommand{\expentry}[4]{%
  % #1 = Job title  #2 = Company | Country  #3 = Dates  #4 = description
  \noindent{\bfseries\color{accent} #1}\hfill{\small\color{muted}\textit{#3}}\\
  {\small\bfseries\color{dark} #2}%
  \ifx&#4&\else
    \vspace{3pt}\begin{itemize}[leftmargin=1.4em, itemsep=2pt, topsep=2pt, parsep=0pt]
      #4
    \end{itemize}
  \fi
  \vspace{7pt}%
}
